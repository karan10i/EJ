# LinkedIn Email Scraper and Sender: Terminal Commands

# 1. Activate your Python virtual environment
source .venv/bin/activate

# 2. Set LinkedIn credentials (for scraping)
export LINKEDIN_USERNAME="your.email@example.com"
export LINKEDIN_PASSWORD="your_password"

# 3. Run the LinkedIn email scraper (example: 3 queries from test_queries.json)
python linkedin_email_scraper.py --queries test_queries.json --max-scrolls 3 --scroll-pause 2.0 --tab-delay 3.0 --output test_emails.csv

# 4. Set Gmail credentials (for sending emails)
export SENDER_EMAIL="karan.gup10@gmail.com"
export SENDER_PASSWORD="your-gmail-app-password"

# 5. Preview emails to be sent (dry run, no emails sent)
python send_emails.py --csv test_emails.csv --resume RESUME2.pdf --dry-run

# 6. Send emails to first 3 contacts (for testing)
python send_emails.py --csv test_emails.csv --resume RESUME2.pdf --limit 3

# 7. Send emails to all contacts (with 10s delay between each)
python send_emails.py --csv test_emails.csv --resume RESUME2.pdf --delay 10
